{{- /* Resolve a list of {pageRef, note?} params entries into {page, note} dicts, failing the build on a broken ref. Input: dict with "refs" (the entry list) and "what" (a label for error messages). The note falls back through the standard description chain. */ -}} {{- $what := .what -}} {{- $resolved := slice -}} {{- range .refs -}} {{- $page := site.GetPage .pageRef -}} {{- if not $page }}{{ errorf "homepage %s page not found: %s" $what .pageRef }}{{ end -}} {{- $resolved = $resolved | append (dict "page" $page "note" (.note | default (partial "description.html" $page))) -}} {{- end -}} {{- return $resolved -}}
